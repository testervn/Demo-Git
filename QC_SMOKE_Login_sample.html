echo "Login to QC"
open /
waitForElementPresent //*[@id='sign-in']
click id=sign-in
pause 3000
echo "Input user name and password"
waitForElementPresent id=email
type id=email yen3@yen.vn
waitForElementPresent id=password
type id=password a1
echo "Submit login form"
clickAndWait name=btnSubmit
pause 5000
echo "Confirm login succeeded"
waitForElementPresent //a[@href='http://ads1v.dev.itim.vn/user/logout']
echo "Restore condition"
clickAndWait //a[@href='http://ads1v.dev.itim.vn/user/logout']
pause 5000
waitForElementNotPresent //a[@href='http://ads1v.dev.itim.vn/user/logout']
